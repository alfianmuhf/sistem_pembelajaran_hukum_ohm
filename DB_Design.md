create table public.admin (
  id_admin bigint generated by default as identity not null,
  email character varying not null,
  password character varying null,
  constraint admin_pkey primary key (id_admin)
) TABLESPACE pg_default;

create table public.guru (
  id_guru bigint generated by default as identity not null,
  nip character varying not null,
  nama_guru character varying not null,
  password character varying not null,
  is_active boolean not null,
  constraint guru_pkey primary key (id_guru),
  constraint guru_nip_key unique (nip)
) TABLESPACE pg_default;

create table public.siswa (
  id_siswa bigint generated by default as identity not null,
  id_kelas bigint null,
  nim character varying not null,
  nama_siswa character varying not null,
  password character varying not null,
  constraint siswa_pkey primary key (id_siswa),
  constraint siswa_id_kelas_fkey foreign KEY (id_kelas) references kelas (id_kelas) on update CASCADE on delete set null
) TABLESPACE pg_default;

create table public.kelas (
  id_kelas bigint generated by default as identity not null,
  id_guru bigint null,
  nama_kelas character varying not null,
  constraint kelas_pkey primary key (id_kelas),
  constraint kelas_kelas_key unique (nama_kelas),
  constraint kelas_id_guru_fkey foreign KEY (id_guru) references guru (id_guru) on update CASCADE on delete set null
) TABLESPACE pg_default;

create table public.sesi (
  id_sesi bigint generated by default as identity not null,
  id_kelas bigint not null,
  id_sesi_sebelum bigint null,
  sesi bigint not null,
  tipe character varying not null,
  tanggal_pembuatan date not null,
  tenggang_waktu timestamp without time zone not null,
  constraint sesi_pkey primary key (id_sesi),
  constraint sesi_id_kelas_fkey foreign KEY (id_kelas) references kelas (id_kelas) on update CASCADE on delete CASCADE,
  constraint sesi_id_sesi_sebelum_fkey foreign KEY (id_sesi_sebelum) references sesi (id_sesi) on update CASCADE on delete CASCADE
) TABLESPACE pg_default;

create table public.soal (
  id_soal bigint generated by default as identity not null,
  id_sesi bigint not null,
  id_siswa bigint not null,
  ohm bigint not null,
  volt bigint not null,
  ampere double precision not null,
  constraint soal_pkey primary key (id_soal),
  constraint soal_id_sesi_fkey foreign KEY (id_sesi) references sesi (id_sesi) on update CASCADE on delete CASCADE,
  constraint soal_id_siswa_fkey foreign KEY (id_siswa) references siswa (id_siswa) on update CASCADE on delete CASCADE
) TABLESPACE pg_default;

create table public.jawaban_soal_siswa (
  id_jawaban_soal_siswa bigint generated by default as identity not null,
  id_soal bigint not null,
  jawaban_soal double precision null,
  constraint jawaban_soal_siswa_pkey primary key (id_jawaban_soal_siswa),
  constraint jawaban_soal_siswa_id_soal_fkey foreign KEY (id_soal) references soal (id_soal) on update CASCADE on delete CASCADE
) TABLESPACE pg_default;

create table public.jawaban_praktikum_siswa (
  id_jawaban_praktikum_siswa bigint generated by default as identity not null,
  id_soal bigint not null,
  volt_sensor double precision null,
  ohm_sensor double precision null,
  ampere_sensor double precision null,
  constraint jawaban_praktikum_siswa_pkey primary key (id_jawaban_praktikum_siswa),
  constraint jawaban_praktikum_siswa_id_soal_fkey foreign KEY (id_soal) references soal (id_soal) on update CASCADE on delete CASCADE
) TABLESPACE pg_default;

create table public.jawaban_analisis_siswa (
  id_jawaban_analisis_siswa bigint generated by default as identity not null,
  id_siswa bigint not null,
  id_sesi bigint not null,
  analisis_siswa character varying null,
  constraint jawaban_analisis_siswa_pkey primary key (id_jawaban_analisis_siswa),
  constraint jawaban_analisis_siswa_id_sesi_fkey foreign KEY (id_sesi) references sesi (id_sesi),
  constraint jawaban_analisis_siswa_id_siswa_fkey foreign KEY (id_siswa) references siswa (id_siswa)
) TABLESPACE pg_default;

create table public.nilai_siswa (
  id_nilai_siswa bigint generated by default as identity not null,

  id_siswa bigint not null,
  nilai_soal bigint null,
  nilai_praktikum bigint null,
